218 void Camera::command(enum camera_command cmd)
219 {
220  switch (cmd)
221  {
222  case CAMERA_MOVE_FORWARD:
223  if (mFlags & Camera_FlyMode)
224  {
225  mPos[2] += (mTranslateDelta * cosf(mTheta));
226  }
227 
228  mPos[0] += (mTranslateDelta * sinf(mTheta));
229  mPos[1] += (mTranslateDelta * sinf(mTheta2));
230  break;
231  case CAMERA_MOVE_BACKWARD:
232  if (mFlags & Camera_FlyMode)
233  {
234  mPos[2] -= (mTranslateDelta * cosf(mTheta));
235  }
236 
237  mPos[0] -= (mTranslateDelta * sinf(mTheta));
238  mPos[1] -= (mTranslateDelta * sinf(mTheta2));
239  break;
240  case CAMERA_MOVE_LEFT:
241  mPos[0] -= (mTranslateDelta * sinf(mTheta - 90.0f));
242  mPos[2] -= (mTranslateDelta * cosf(mTheta - 90.0f));
243  break;
244  case CAMERA_MOVE_RIGHT:
245  mPos[0] -= (mTranslateDelta * sinf(mTheta + 90.0f));
246  mPos[2] -= (mTranslateDelta * cosf(mTheta + 90.0f));
247  break;
248  case CAMERA_ROTATE_UP:
249  if (mTheta2 < (M_PI / 2)) {
250  mTheta2 += mRotateDelta2;
251  rotate(mTheta2, 1.0f, 0.0f, 0.0f);
252  }
253  break;
254  case CAMERA_ROTATE_DOWN:
255  if (mTheta2 > -(M_PI / 2)) {
256  mTheta2 -= mRotateDelta2;
257  rotate(mTheta2, 1.0f, 0.0f, 0.0f);
258  }
259  break;
260  case CAMERA_ROTATE_RIGHT:
261  mTheta += mRotateDelta;
262  rotate(mTheta, 0.0f, 1.0f, 0.0f);
263  break;
264  case CAMERA_ROTATE_LEFT:
265  mTheta -= mRotateDelta;
266  rotate(mTheta, 0.0f, 1.0f, 0.0f);
267  break;
268  case CAMERA_MOVE_UP:
269  mPos[1] -= mTranslateDelta / 2.0f;
270  mTarget[1] -= mTranslateDelta / 2.0f;
271  break;
272  case CAMERA_MOVE_DOWN:
273  mPos[1] += mTranslateDelta / 2.0f;
274  mTarget[1] += mTranslateDelta / 2.0f;
275  break;
276  case CAMERA_SPEED_UP:
277  ++mTranslateDelta;
278  break;
279  case CAMERA_SPEED_DOWN:
280  if (--mTranslateDelta < 0.0f)
281  mTranslateDelta = 1.0f;
282  break;
283  default:
284  ;
285  }
286 }

290 bool Camera::isBehind(int x, int z)
291 {

294 
295  // Set up a "virtual camera" a huge distance behind us
296  bTheta = mTheta + HEL_PI;
297

299  bTheta -= HEL_2_PI;
300 
301  // 64k is a fair distance away...
