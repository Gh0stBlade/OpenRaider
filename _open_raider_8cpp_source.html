561  assert(mInit == true);

563 
564  pakDir = opendir(dir);
565  if (pakDir != NULL) {
566  while ((ep = readdir(pakDir)) != NULL) {
567  if (ep->d_type == DT_DIR) {
568  if ((strcmp(".", ep->d_name) != 0)
569  && (strcmp("..", ep->d_name) != 0)) {
570  char *tmp = bufferString("%s%s", dir, ep->d_name);
571  char *next = fullPath(tmp, '/');
572  loadPakFolderRecursive(next);
573  delete next;
574  delete tmp;
575  }
576  } else {
577  char *fullPathMap = bufferString("%s%s", dir, ep->d_name);
578 
579  char *lowerPath = bufferString("%s", fullPathMap);
580  for (char *p = lowerPath; *p; ++p) *p = (char)tolower(*p);
581 
582  // Check for valid extension
583  if (stringEndsWith(lowerPath, ".phd")
584  || stringEndsWith(lowerPath, ".tr2")
585  || stringEndsWith(lowerPath, ".tr4")
586  || stringEndsWith(lowerPath, ".trc")) {
587  int error = TombRaider::checkMime(fullPathMap);
588  if (error == 0) {
589  // Just load relative filename

591  } else {

594  }

649  assert(mInit == true);
650  assert(mRunning == false);
651 
652  static clock_t fpsSum = 0, fpsCount = 0;
653  static int fps = 0;
654 
655  mRunning = true;
656  while (mRunning) {
657  clock_t startTime = systemTimerGet();
658 
659  // Get keyboard and mouse input
660  mWindow->eventHandling();
661 
662  // Clear screen
663  glClearColor(0.00f, 0.00f, 0.00f, 1.0f);
664  glClear(GL_COLOR_BUFFER_BIT);
665 
666  // Draw game scene
667  mGame->display();
668 
669  // Draw 2D overlays (console and menu)
670  mWindow->glEnter2D();
671 
672  mConsole->display();
673  mMenu->display();
674 
675  // Draw FPS counter
676  if (mFPS)
677  mWindow->drawText(10, mWindow->mHeight - 20, 0.5f, OR_BLUE, "%dFPS", fps);
678 
679  mWindow->glExit2D();
680 
681  // Put new frame on screen
682  mWindow->swapBuffersGL();
683 
684  // Fill map list after first render pass,
685  // so menu *loading screen* is visible
686  if (!mMapListFilled)
687  fillMapList();
688 
689  // Calculate FPS display value
690  fpsCount++;
691  fpsSum += (systemTimerGet() - startTime);
692  if (fpsSum >= 500) {
693  // Update every 500ms
694  fps = (int)((float)fpsCount * (1000.0f / (float)fpsSum));
695  fpsCount = fpsSum = 0;
696  }
697  }
698 }
