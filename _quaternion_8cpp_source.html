158 Quaternion Quaternion::slerp(Quaternion a, Quaternion b, vec_t time) {
159  /*******************************************************************
160  * Spherical Linear Interpolation algorthim
161  *-----------------------------------------------------------------
162  *
163  * Interpolate between A and B rotations ( Find qI )
164  *
165  * qI = (((qB . qA)^ -1)^ Time) qA
166  *
167  * http://www.magic-software.com/Documentation/quat.pdf
168  *
169  * Thanks to digiben for algorithms and basis of the notes in
170  * this func
171  *
172  *******************************************************************/
173 
174  vec_t result, scaleA, scaleB;
175  Quaternion i;
176 
177 
178  // Don't bother if it's the same rotation, it's the same as the result
179  if (a == b)
180  return a;
181 
182  // A . B
183  result = dot(a, b);
184 
185  // If the dot product is less than 0, the angle is greater than 90 degrees
186  if (result < 0.0f) {
187  // Negate quaternion B and the result of the dot product
188  b = Quaternion(-b.mW, -b.mX, -b.mY, -b.mZ);
189  result = -result;
190  }
191 
192  // Set the first and second scale for the interpolation
193  scaleA = 1 - time;
194  scaleB = time;
195 
196  // Next, we want to actually calculate the spherical interpolation. Since this
197  // calculation is quite computationally expensive, we want to only perform it
198  // if the angle between the 2 quaternions is large enough to warrant it. If the
199  // angle is fairly small, we can actually just do a simpler linear interpolation
200  // of the 2 quaternions, and skip all the complex math. We create a "delta" value
201  // of 0.1 to say that if the cosine of the angle (result of the dot product) between
202  // the 2 quaternions is smaller than 0.1, then we do NOT want to perform the full on
203  // interpolation using. This is because you won't really notice the difference.
204 
205  // Check if the angle between the 2 quaternions was big enough
206  // to warrant such calculations
207  if (1 - result > 0.1f) {
208  // Get the angle between the 2 quaternions, and then
209  // store the sin() of that angle
210  vec_t theta = (float)acos(result);
211  vec_t sinTheta = (float)sin(theta);
212 
213  // Calculate the scale for qA and qB, according to
214  // the angle and it's sine value
215  scaleA = (float)sin((1 - time) * theta) / sinTheta;
216  scaleB = (float)sin((time * theta)) / sinTheta;
217  }
218 
219  // Calculate the x, y, z and w values for the quaternion by using a special
220  // form of linear interpolation for quaternions.
221  return (a.scale(scaleA) + b.scale(scaleB));
222 }

224 void Quaternion::setByMatrix(matrix_t matrix) {
225  float diagonal = matrix[0] + matrix[5] + matrix[10] + 1.0f;
226  float scale = 0.0f;
227  float w = 0.0f, x = 0.0f, y = 0.0f, z = 0.0f;
228 
229  if (diagonal > 0.00000001) {
230  // Calculate the scale of the diagonal
231  scale = (float)(sqrt(diagonal) * 2);
232 
233  w = 0.25f * scale;
234  x = (matrix[9] - matrix[6]) / scale;
235  y = (matrix[2] - matrix[8]) / scale;
236  z = (matrix[4] - matrix[1]) / scale;
237  } else {
238  // If the first element of the diagonal is the greatest value
239  if (matrix[0] > matrix[5] && matrix[0] > matrix[10]) {
240  // Find the scale according to the first element, and double it
241  scale = (float)sqrt(1.0f + matrix[0] - matrix[5] - matrix[10])*2.0f;
242 
243  // Calculate the quaternion

245  x = 0.25f * scale;
246  y = (matrix[4] + matrix[1]) / scale;
247  z = (matrix[2] + matrix[8]) / scale;
248  } else if (matrix[5] > matrix[10]) {
249  // The second element of the diagonal is the greatest value
250  // Find the scale according to the second element, and double it
251  scale = (float)sqrt(1.0f + matrix[5] - matrix[0] - matrix[10])*2.0f;
252

255  x = (matrix[4] + matrix[1]) / scale;

258  } else { // The third element of the diagonal is the greatest value
