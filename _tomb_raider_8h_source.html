335 /*--------------------------------------------------------------
336  * Room sector structure.
337  *
338  * Sectors are 1024 * 1024 (world coordinates). Floor and
339  * Ceiling are signed number of 256 units of height
340  * (relative to 0), e.g. Floor 0x04 corresponds to
341  * Y = 1024 in world coordinates.
342  *
343  * Note: this implies that, while X and Z can be quite large,
344  * Y is constrained to -32768..32512. Floor/Ceiling value of
345  * 0x81 indicates impenetrable wall.
346  *
347  * Floor values are used by the game engine to determine what
348  * objects Lara can traverse and how.
349  *
350  * Relative steps of 1 (-256) can be walked up;
351  * steps of 2..7 (-512..-1792) can/must be climbed;
352  * steps larger than 7 (-2048..-32768) cannot be climbed (too tall)
353  *
354  * In TR3, BoxIndex is more complicated. Only bits 4-14 are the
355  * "real" index; bits 0-3 are most likely some kind of flag,
356  * such as what kind of footstep sound to make (wood, metal, snow).
357  * Furthermore, there is a special value of the "real" index,
358  * 2047, or 0x7ff.
359  *
360  * RoomAbove and RoomBelow indicate what neighboring rooms are
361  * in these directions; if RoomAbove is not "none", then the
362  * ceiling is a collisional portal to that room, while if
363  * RoomBelow is not "none", then the floor is a collisional
364  * portal to that room.
365  --------------------------------------------------------------*/
366 typedef struct tr2_room_sector_s
367 {
368  unsigned short fd_index; // Index into FloorData[]
369  unsigned short box_index; // Index into Boxes[]/Zones[] ( -1 if none )
370  unsigned char room_below; // The number of the room below this one
371  // ( -1 or 255 if none )
372  char floor; // Absolute height of floor
373  // ( Multiply by 256 for world coordinates )
374  unsigned char room_above; // The number of the room above this one
375  // ( -1 or 255 if none )
376  char ceiling; // Absolute height of ceiling
377  // ( Multiply by 256 for world coordinates )
378 } __attribute__ ((packed)) tr2_room_sector_t;

757 /*--------------------------------------------------------------
758  * Frame structure.
759  *
760  * Frames indicates how composite meshes are positioned and rotated.
761  * They work in conjunction with Animations[] and Bone2[].
762  *
763  * A given frame has the following format:
764  * short BB1x, BB1y, BB1z // bounding box (low)
765  * short BB2x, BB2y, BB2z // bounding box (high)
766  * short OffsetX, OffsetY, OffsetZ // starting offset for this moveable

768  * (TR2/3: NumValues is implicitly NumMeshes (from moveable))

770  * What follows next is a list of angle sets. In TR2/3, an angle set can
771  * specify either one or three axes of rotation. If either of the high two
772  * bits (0xc000) of the first angle unsigned short are set, it's one axis:
773  * only one unsigned short,
774  * low 10 bits (0x03ff),
775  * scale is 0x100 == 90 degrees;
776  * the high two bits are interpreted as follows:
777  * 0x4000 == X only, 0x8000 == Y only,
778  * 0xC000 == Z only.
779  *
780  * If neither of the high bits are set, it's a three-axis rotation. The next
781  * 10 bits (0x3ff0) are the X rotation, the next 10 (including the following
782  * unsigned short) (0x000f, 0xfc00) are the Y rotation,
783  * the next 10 (0x03ff) are the Z rotation, same scale as
784  * before (0x100 == 90 degrees).
785  *
786  * Rotations are performed in Y, X, Z order.
787  * TR1 ONLY: All angle sets are two words and interpreted like the two-word
788  * sets in TR2/3, EXCEPT that the word order is reversed.
789  --------------------------------------------------------------*/
