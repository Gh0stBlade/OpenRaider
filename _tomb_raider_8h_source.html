334 /*--------------------------------------------------------------
335  * Room sector structure.
336  *
337  * Sectors are 1024 * 1024 (world coordinates). Floor and
338  * Ceiling are signed number of 256 units of height
339  * (relative to 0), e.g. Floor 0x04 corresponds to
340  * Y = 1024 in world coordinates.
341  *
342  * Note: this implies that, while X and Z can be quite large,
343  * Y is constrained to -32768..32512. Floor/Ceiling value of
344  * 0x81 indicates impenetrable wall.
345  *
346  * Floor values are used by the game engine to determine what
347  * objects Lara can traverse and how.
348  *
349  * Relative steps of 1 (-256) can be walked up;
350  * steps of 2..7 (-512..-1792) can/must be climbed;
351  * steps larger than 7 (-2048..-32768) cannot be climbed (too tall)
352  *
353  * In TR3, BoxIndex is more complicated. Only bits 4-14 are the
354  * "real" index; bits 0-3 are most likely some kind of flag,
355  * such as what kind of footstep sound to make (wood, metal, snow).
356  * Furthermore, there is a special value of the "real" index,
357  * 2047, or 0x7ff.
358  *
359  * RoomAbove and RoomBelow indicate what neighboring rooms are
360  * in these directions; if RoomAbove is not "none", then the
361  * ceiling is a collisional portal to that room, while if
362  * RoomBelow is not "none", then the floor is a collisional
363  * portal to that room.
364  --------------------------------------------------------------*/
365 typedef struct tr2_room_sector_s
366 {
367  unsigned short fd_index; // Index into FloorData[]
368  unsigned short box_index; // Index into Boxes[]/Zones[] ( -1 if none )
369  unsigned char room_below; // The number of the room below this one
370  // ( -1 or 255 if none )
371  char floor; // Absolute height of floor
372  // ( Multiply by 256 for world coordinates )
373  unsigned char room_above; // The number of the room above this one
374  // ( -1 or 255 if none )
375  char ceiling; // Absolute height of ceiling
376  // ( Multiply by 256 for world coordinates )
377 } __attribute__ ((packed)) tr2_room_sector_t;

756 /*--------------------------------------------------------------
757  * Frame structure.
758  *
759  * Frames indicates how composite meshes are positioned and rotated.
760  * They work in conjunction with Animations[] and Bone2[].
761  *
762  * A given frame has the following format:
763  * short BB1x, BB1y, BB1z // bounding box (low)
764  * short BB2x, BB2y, BB2z // bounding box (high)

770  * specify either one or three axes of rotation. If either of the high two

772  * only one unsigned short,
773  * low 10 bits (0x03ff),

780  * 10 bits (0x3ff0) are the X rotation, the next 10 (including the following
781  * unsigned short) (0x000f, 0xfc00) are the Y rotation,
782  * the next 10 (0x03ff) are the Z rotation, same scale as
783  * before (0x100 == 90 degrees).
784  *
785  * Rotations are performed in Y, X, Z order.
786  * TR1 ONLY: All angle sets are two words and interpreted like the two-word
