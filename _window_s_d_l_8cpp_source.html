552  color.a = (unsigned char)(s->color[3] * 255.0f);
553 
554  SDL_Surface *surface = TTF_RenderUTF8_Blended(mFont, s->text, color);
555  if (surface == NULL) {
556  printf("TTF_RenderUTF8_Blended Error: %s\n", TTF_GetError());

558  }
559 
560  s->w = (int)((float)surface->w * s->scale);
561  s->h = (int)((float)surface->h * s->scale);
562 
563  GLenum textureFormat;
564  if (surface->format->BytesPerPixel == 4) {
565  if (surface->format->Rmask == 0x000000FF)
566  textureFormat = GL_RGBA;
567  else
568  textureFormat = GL_BGRA_EXT;
569  } else {
570  if (surface->format->Rmask == 0x000000FF)
571  textureFormat = GL_RGB;
572  else
573  textureFormat = GL_BGR;
574  }
575 
576  glBindTexture(GL_TEXTURE_2D, mFontTexture);
577  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
578  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
579  glTexImage2D(GL_TEXTURE_2D, 0, surface->format->BytesPerPixel, surface->w, surface->h, 0, textureFormat, GL_UNSIGNED_BYTE, surface->pixels);
580 
581  GLuint xMin = s->x;
582  GLuint yMin = s->y;

584  GLuint yMax = yMin + (int)((float)surface->h * s->scale);
585

587
