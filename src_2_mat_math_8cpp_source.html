88 
89 
90  a = (square(posB[0] - posA[0]) +
91  square(posB[1] - posA[1]) +
92  square(posB[2] - posA[2]));
93  b = (2 * ((posB[0] - posA[0]) * (posA[0] - center[0]) +
94  (posB[1] - posA[1]) * (posA[1] - center[1]) +
95  (posB[2] - posA[2]) * (posA[2] - center[2])));
96  c = (square(center[0]) + square(center[1]) +
97  square(center[2]) + square(posA[0]) +

103 
104 
105  if (i < 0.0)

107  // No intersection
