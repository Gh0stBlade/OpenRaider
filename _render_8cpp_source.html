317 void gluLookAt(float eyeX, float eyeY, float eyeZ,
318  float lookAtX, float lookAtY, float lookAtZ,
319  float upX, float upY, float upZ) {
320  float f[3];
321 
322  // calculating the viewing vector
323  f[0] = lookAtX - eyeX;
324  f[1] = lookAtY - eyeY;
325  f[2] = lookAtZ - eyeZ;
326 
327  float fMag = sqrtf(f[0] * f[0] + f[1] * f[1] + f[2] * f[2]);
328  //float upMag = sqrtf(upX * upX + upY * upY + upZ * upZ);
329 
330  // normalizing the viewing vector
331  f[0] = f[0] / fMag;
332  f[1] = f[1] / fMag;
333  f[2] = f[2] / fMag;
334 
335  float s[3] = {
336  f[1] * upZ - upY * f[2],
337  upX * f[2] - f[0] * upZ,
338  f[0] * upY - upX * f[1]
339  };
340 
341  float u[3] = {
342  s[1] * f[2] - f[1] * s[2],
343  f[0] * s[2] - s[0] * f[2],
344  s[0] * f[1] - f[0] * s[1]
345  };
346 
347  float m[16] = {
348  s[0], u[0], -f[0], 0,
349  s[1], u[1], -f[1], 0,
350  s[2], u[2], -f[2], 0,
351  0, 0, 0, 1
352  };
353  glMultMatrixf(m);
354  glTranslatef(-eyeX, -eyeY, -eyeZ);
355 }
