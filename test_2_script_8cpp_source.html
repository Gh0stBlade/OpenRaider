134  // Initialize decompression
135  z_stream stream;

142  return 1;

144 
145  // Inflate data in one go
146  stream.avail_in = size;
147  stream.next_in = const_cast<unsigned char *>(data);
148  stream.avail_out = bufferSize;
149  stream.next_out = buffer;
150  error = inflate(&stream, Z_FINISH);

152  std::cout << "inflate() Error " << error << std::endl;
153  return 2;
154  }

156 
157  // Write buffer to file
158  std::ofstream s(file, std::ios_base::out | std::ios_base::binary);
159  s.write(reinterpret_cast<const char *>(buffer), bufferSize - stream.avail_out);
160 
161  return 0;
