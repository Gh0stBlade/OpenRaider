165 void Menu::displayMapList() {
166  // Estimate displayable number of items
167  int items = (getWindow().getHeight() - 110) / 25;
168 
169  // Select which part of the list to show
170  int min, max;
171  if (((int)mCursor - (items / 2)) > 0)
172  min = mCursor - (items / 2);
173  else
174  min = 0;
175 
176  if ((mCursor + (items / 2)) < mMapList.size())
177  max = mCursor + (items / 2);
178  else
179  max = mMapList.size();
180 
181  while ((max - min) < items) {
182  if (min > 0)
183  min--;
184  else if (max < ((int)mMapList.size()))
185  max++;
186  else
187  break;
188  }
189 
190  mMin = min;
191 
192  for (int i = 0; i < (max - min); i++) {
193  char *map = mMapList[i + min];
194  if ((i + min) == (int)mCursor)
195  getFont().drawText(25, 100 + (25 * i), 0.75f, RED, "%s", map);
196  else
