60 void Menu::loadPakFolderRecursive(const char *dir) {
61  struct dirent entry;
62  struct dirent *ep = NULL;
63  DIR *pakDir;
64 
65  assert(dir != NULL);
66  assert(dir[0] != '\0');
67 
68  pakDir = opendir(dir);
69  if (pakDir != NULL) {
70  readdir_r(pakDir, &entry, &ep);
71  while (ep != NULL) {
72  if (ep->d_type == DT_DIR) {
73  if ((strcmp(".", ep->d_name) != 0)
74  && (strcmp("..", ep->d_name) != 0)) {
75  char *tmp = bufferString("%s%s", dir, ep->d_name);
76  char *next = fullPath(tmp, '/');
77  loadPakFolderRecursive(next);
78  delete next;
79  delete tmp;
80  }
81  } else {
82  char *fullPathMap = bufferString("%s%s", dir, ep->d_name);
83 
84  char *lowerPath = bufferString("%s", fullPathMap);
85  for (char *p = lowerPath; *p; ++p) *p = (char)tolower(*p);
86 
87  // Check for valid extension
88  if (stringEndsWith(lowerPath, ".phd")
89  || stringEndsWith(lowerPath, ".tr2")
90  || stringEndsWith(lowerPath, ".tr4")
91  || stringEndsWith(lowerPath, ".trc")) {
92  int error = TombRaider::checkMime(fullPathMap);
93  if (error == 0) {
94  // Just load relative filename
95  mMapList.push_back(bufferString("%s", (fullPathMap + strlen(getOpenRaider().mPakDir) + 1)));
96  } else {
97  getConsole().print("Error: pak file '%s' %s",
98  fullPathMap, (error == -1) ? "not found" : "invalid");
99  }
100  }
101 
102  delete [] lowerPath;
103  delete [] fullPathMap;
104  }
105  readdir_r(pakDir, &entry, &ep);
106  }
107  closedir(pakDir);
108  } else {
109  getConsole().print("Could not open PAK dir %s!", dir);
110  }
